```PHP
<?php
 $ip=$_SERVER['REMOTE_ADDR'];
 $subject="Acceso a secreto";
 $message = "Name: " . $message . "IP of sender: " . $ip;
 mail( "info@example.com", $subject, $message, "From: info@example.com" );
?>
```
