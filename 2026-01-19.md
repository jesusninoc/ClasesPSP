# SHA-512
```Kotlin
import java.security.MessageDigest

fun sha256Hex(texto: String): String {
    val digest = MessageDigest.getInstance("SHA-256")
    val hashedBytes = digest.digest(texto.toByteArray(Charsets.UTF_8))
    return hashedBytes.joinToString("") { "%02x".format(it) }
}

// Ejemplo de uso
fun main() {
    val texto = "Hola, mundo!"
    println("Hash SHA-256: ${sha256Hex(texto)}")
}
```

# HMAC-MD5 y SHA-512
```Kotlin
import javax.crypto.Mac
import javax.crypto.spec.SecretKeySpec
import java.security.MessageDigest

fun toHex(bytes: ByteArray): String =
    bytes.joinToString("") { "%02x".format(it) }

fun sha256Hex(text: String): String =
    toHex(MessageDigest.getInstance("SHA-256").digest(text.toByteArray()))

fun hmacMD5Hex(key: String, text: String): String {
    val mac = Mac.getInstance("HmacMD5")
    mac.init(SecretKeySpec(key.toByteArray(), "HmacMD5"))
    return toHex(mac.doFinal(text.toByteArray()))
}

fun main() {
    val key = "mySecretKey"
    val plainText = "Hola, mundo!"

    println("Texto plano: $plainText")
    println("HMAC-MD5: ${hmacMD5Hex(key, plainText)}")
    println("SHA-256: ${sha256Hex(plainText)}")
}
```

# AES (CBC)
```Kotlin
import java.security.Key
import java.util.Base64
import javax.crypto.Cipher
import javax.crypto.KeyGenerator
import javax.crypto.spec.IvParameterSpec

fun generateAESKey(): Key =
    KeyGenerator.getInstance("AES").apply { init(128) }.generateKey()

fun encryptAES(text: String, key: Key): ByteArray {
    val cipher = Cipher.getInstance("AES/CBC/PKCS5Padding")
    cipher.init(Cipher.ENCRYPT_MODE, key)
    val iv = cipher.iv                      // IV ya generado
    val enc = cipher.doFinal(text.toByteArray())
    return iv + enc                         // guardo IV + cifrado juntos
}

fun decryptAES(data: ByteArray, key: Key): String {
    val iv = data.copyOfRange(0, 16)
    val enc = data.copyOfRange(16, data.size)
    val cipher = Cipher.getInstance("AES/CBC/PKCS5Padding")
    cipher.init(Cipher.DECRYPT_MODE, key, IvParameterSpec(iv))
    return String(cipher.doFinal(enc))
}

fun main() {
    val msg = "Hola!"
    val key = generateAESKey()

    val encrypted = encryptAES(msg, key)
    val decrypted = decryptAES(encrypted, key)

    println("Original: $msg")
    println("Cifrado (Base64): ${Base64.getEncoder().encodeToString(encrypted)}")
    println("Descifrado: $decrypted")
}
```

# RSA (.der)
```Kotlin
import java.security.*
import java.security.spec.PKCS8EncodedKeySpec
import java.security.spec.X509EncodedKeySpec
import java.nio.file.Files
import java.nio.file.Paths
import java.util.Base64

fun main() {
    val (pub, priv) = generateRSA()

    saveBytes("public_key.der", pub.encoded)
    saveBytes("private_key.der", priv.encoded)

    val pub2 = loadPublic("public_key.der")
    val priv2 = loadPrivate("private_key.der")

    println("Public (Base64): ${Base64.getEncoder().encodeToString(pub2.encoded)}")
    println("Private (Base64): ${Base64.getEncoder().encodeToString(priv2.encoded)}")
}

fun generateRSA(): KeyPair =
    KeyPairGenerator.getInstance("RSA").apply { initialize(2048) }.generateKeyPair()

fun saveBytes(file: String, bytes: ByteArray) =
    Files.write(Paths.get(file), bytes)

fun loadPublic(file: String): PublicKey {
    val bytes = Files.readAllBytes(Paths.get(file))
    return KeyFactory.getInstance("RSA").generatePublic(X509EncodedKeySpec(bytes))
}

fun loadPrivate(file: String): PrivateKey {
    val bytes = Files.readAllBytes(Paths.get(file))
    return KeyFactory.getInstance("RSA").generatePrivate(PKCS8EncodedKeySpec(bytes))
}
```

# RSA (.pem)
```Kotlin
import java.nio.file.Files
import java.nio.file.Paths
import java.security.KeyFactory
import java.security.KeyPair
import java.security.KeyPairGenerator
import java.security.PrivateKey
import java.security.PublicKey
import java.security.spec.PKCS8EncodedKeySpec
import java.security.spec.X509EncodedKeySpec
import java.util.Base64

fun main() {
    val keyPair = generateRSA()

    // Guardar en PEM (texto)
    savePem("public_key.pem", "PUBLIC KEY", keyPair.public.encoded)
    savePem("private_key.pem", "PRIVATE KEY", keyPair.private.encoded)

    // Cargar desde PEM (texto)
    val publicKey = loadPublicKeyPem("public_key.pem")
    val privateKey = loadPrivateKeyPem("private_key.pem")

    println("OK - Clave pública cargada: ${publicKey.algorithm}")
    println("OK - Clave privada cargada: ${privateKey.algorithm}")
}

fun generateRSA(): KeyPair =
    KeyPairGenerator.getInstance("RSA").apply { initialize(2048) }.generateKeyPair()

fun savePem(fileName: String, type: String, derBytes: ByteArray) {
    val encoder = Base64.getMimeEncoder(64, "\n".toByteArray())
    val b64 = encoder.encodeToString(derBytes)
    val pem = "-----BEGIN $type-----\n$b64\n-----END $type-----\n"
    Files.writeString(Paths.get(fileName), pem)
}

fun readPemBytes(fileName: String, type: String): ByteArray {
    val pem = Files.readString(Paths.get(fileName))
    val header = "-----BEGIN $type-----"
    val footer = "-----END $type-----"
    val base64 = pem
        .replace(header, "")
        .replace(footer, "")
        .replace(Regex("\\s"), "")   // quita saltos de línea y espacios
    return Base64.getDecoder().decode(base64)
}

fun loadPublicKeyPem(fileName: String): PublicKey {
    val bytes = readPemBytes(fileName, "PUBLIC KEY")
    val spec = X509EncodedKeySpec(bytes)
    return KeyFactory.getInstance("RSA").generatePublic(spec)
}

fun loadPrivateKeyPem(fileName: String): PrivateKey {
    val bytes = readPemBytes(fileName, "PRIVATE KEY")
    val spec = PKCS8EncodedKeySpec(bytes)
    return KeyFactory.getInstance("RSA").generatePrivate(spec)
}
```

# Sockets seguros
## Servidor
```Kotlin
import java.io.DataInputStream
import java.io.DataOutputStream
import javax.net.ssl.SSLServerSocketFactory

fun main() {
    val puerto = 5556

    // Java usa este JKS para la clave/cert del servidor
    System.setProperty("javax.net.ssl.keyStore", "AlmacenSrv.jks")
    System.setProperty("javax.net.ssl.keyStorePassword", "1234567")

    val serverFactory = SSLServerSocketFactory.getDefault() as SSLServerSocketFactory
    val server = serverFactory.createServerSocket(puerto)

    println("Servidor SSL escuchando en $puerto...")

    val socket = server.accept()
    val entrada = DataInputStream(socket.getInputStream())
    val salida = DataOutputStream(socket.getOutputStream())

    println("Cliente dice: ${entrada.readUTF()}")
    salida.writeUTF("Hola desde el servidor SSL")

    entrada.close()
    salida.close()
    socket.close()
    server.close()
}
```

## Cliente
```Kotlin
import java.io.DataInputStream
import java.io.DataOutputStream
import javax.net.ssl.SSLSocketFactory

fun main() {
    val host = "localhost"
    val puerto = 5556

    // Para que el cliente confíe, le damos un trustStore (aquí el mismo JKS, por simplicidad)
    System.setProperty("javax.net.ssl.trustStore", "AlmacenSrv.jks")
    System.setProperty("javax.net.ssl.trustStorePassword", "1234567")

    val socketFactory = SSLSocketFactory.getDefault() as SSLSocketFactory
    val socket = socketFactory.createSocket(host, puerto)

    val salida = DataOutputStream(socket.getOutputStream())
    val entrada = DataInputStream(socket.getInputStream())

    salida.writeUTF("Hola desde el cliente SSL")
    println("Servidor dice: ${entrada.readUTF()}")

    entrada.close()
    salida.close()
    socket.close()
}
```
