# Mail
```Kotlin
dependencies {
    implementation("org.simplejavamail:simple-java-mail:8.12.6")
}

import org.simplejavamail.email.EmailBuilder
import org.simplejavamail.mailer.MailerBuilder
import org.simplejavamail.mailer.config.TransportStrategy

fun main() {
    val host = "smtp.example.com"
    val port = 587
    val user = "tucorreo@example.com"
    val pass = "tupassword"

    val email = EmailBuilder.startingBlank()
        .from(user)
        .to("destinatario@example.com")
        .withSubject("Ejemplo de correo con Kotlin")
        .withPlainText("Hola! Email enviado desde Kotlin.")
        .buildEmail()

    MailerBuilder
        .withSMTPServer(host, port, user, pass)
        .withTransportStrategy(TransportStrategy.SMTP_TLS)
        .buildMailer()
        .sendMail(email)
}
```

# SSH
```Kotlin
import net.schmizz.sshj.SSHClient

fun main() {
    val host = "tu.servidor.ssh.com"
    val user = "tu_usuario_ssh"
    val pass = "tu_contraseña_ssh"
    val command = "ls -l"

    SSHClient().use { ssh ->
        ssh.addHostKeyVerifier { _, _, _ -> true } // INSEGURO (solo para pruebas)
        ssh.connect(host)
        ssh.authPassword(user, pass)

        ssh.startSession().use { session ->
            val cmd = session.exec(command)
            val output = cmd.inputStream.bufferedReader().readText()
            cmd.join()
            println(output)
        }
    }
}
```

# SHA-512
```Kotlin
import java.security.MessageDigest

fun sha256Hex(texto: String): String {
    val digest = MessageDigest.getInstance("SHA-256")
    val hashedBytes = digest.digest(texto.toByteArray(Charsets.UTF_8))
    return hashedBytes.joinToString("") { "%02x".format(it) }
}

// Ejemplo de uso
fun main() {
    val texto = "Hola, mundo!"
    println("Hash SHA-256: ${sha256Hex(texto)}")
}
```

# HMAC-MD5 y SHA-512
```Kotlin
import javax.crypto.Mac
import javax.crypto.spec.SecretKeySpec
import java.security.MessageDigest

fun toHex(bytes: ByteArray): String =
    bytes.joinToString("") { "%02x".format(it) }

fun sha256Hex(text: String): String =
    toHex(MessageDigest.getInstance("SHA-256").digest(text.toByteArray()))

fun hmacMD5Hex(key: String, text: String): String {
    val mac = Mac.getInstance("HmacMD5")
    mac.init(SecretKeySpec(key.toByteArray(), "HmacMD5"))
    return toHex(mac.doFinal(text.toByteArray()))
}

fun main() {
    val key = "mySecretKey"
    val plainText = "Hola, mundo!"

    println("Texto plano: $plainText")
    println("HMAC-MD5: ${hmacMD5Hex(key, plainText)}")
    println("SHA-256: ${sha256Hex(plainText)}")
}
```
